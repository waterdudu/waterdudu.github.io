以下调用合法

  • 调用 pointer receiver method，只要调用者是 pointer type，或者它能被取地址（&）
  • 调用 value recieiver method，只要调用者是 value type，或者它的 value 可以被取值（*）
type List [] int

func (l List) Len() int { return len(l) }
func (l *List) Append(val int) { *l = append(*l, val) }

// call
var list List
list.Append(l)
fmt.Printf("%v (len: %d) \n", list, list.Len())

pList := new(List)
pList.Append(2)
fmt.Printf("%v (len: %d) \n", pList, pList.Len())

for range string

for range 默认会把 string 当作 utf-8 字符，如果不能解析，则返回 0xfffd。 需要把它转换成 byte slice 让结果正确。

    data := "A\xfe\x02\xff\x04"
    for _,v := range data {
        fmt.Printf("%#x ",v)
    }
    //prints: 0x41 0xfffd 0x2 0xfffd 0x4 (not ok)

    fmt.Println()
    for _,v := range []byte(data) {
        fmt.Printf("%#x ",v)
    }
    //prints: 0x41 0xfe 0x2 0xff 0x4 (good)

for range map 随机

    m := map[string]int{"one": 1, "two": 2, "three": 3, "four": 4}
    for k, v := range m {
        fmt.Println(k, v)
    }

    // first time run
    
    // [ `run` | done: 9.628665ms ]
    //  one 1
    //  two 2
    //  three 3
    //  four 4


    // second time run

    // [ `run` | done: 8.847101ms ]
    //  three 3
    //  four 4
    //  one 1
    //  two 2

switch

switch 默认是 fallthrough 的，可以在 case 语句里写成 list，或者使用 fallthrough 语句

    isSpace := func(ch byte) bool {
        switch(ch) {
        case ' ', '\t':
            return true
        }
        return false
    }

未导出

未导出的 field 不能被 encode，decode 的时候被初始化为 zero value

package main

import (  
    "fmt"
    "encoding/json"
)

type MyData struct {  
    One int
    two string
}

func main() {  
    in := MyData{1,"two"}
    fmt.Printf("%#v\n",in) //prints main.MyData{One:1, two:"two"}

    encoded,_ := json.Marshal(in)
    fmt.Println(string(encoded)) //prints {"One":1}

    var out MyData
    json.Unmarshal(encoded,&out)

    fmt.Printf("%#v\n",out) //prints main.MyData{One:1, two:""}
}

template

template 加载可以有多种方式:

  • Parse
  • ParseFiles

简单


type Tag struct {
    TName string
}
type Person struct {
    FirstName string
    LastName  string
    Age       int
    Emails    []string
    Tags      []*Tag
}

func main() {
    t1 := Tag{TName: "java"}
    t2 := Tag{TName: "go"}
    t := template.New("persion tag example")
    t, _ = t.Parse(`person with tags {{.FirstName}} !
        {{range .Emails}}
            email : {{.}}
        {{end}}
        {{with .Tags}}
            {{range .}}
                tag is : {{.TName}}
            {{end}}
        {{end}}
    `)
    p := Person{
        FirstName: "d",
        LastName:  "X",
        Emails:    []string{"a@gmail.com", "b@gmail.com"},
        Tags:      []*Tag{&t1, &t2},
    }

    t.Execute(os.Stdout, p)
}

output

person with tags d !
            email : a@gmail.com
            email : b@gmail.com
                tag is : java
                tag is : go

组合 template

把一个页面分成 layout 和 content，layout 负责布局， content 负责展示内容

content.gtpl 定义了 template 名字：content

{{define "content"}}
<ul>
    {{range .}}
        <li>{{.FirstName}} {{.LastName}} is {{.Age}} years old.
        {{.FirstName}}'s full name is {{printf "%s%s" .FirstName .LastName | len}} letters long.
        In 10 years, {{.FirstName}} will be {{add .Age 10}}.</li>
    {{end}}
</ul>
{{end}}

layout.gtpl 使用 content.gtpl 定义的 content template

<div class="outer">
    <h1>{{.Title}}</h1>
    {{template "content" .Content}}
</div>
